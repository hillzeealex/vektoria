Quickstart

Up and running in a few lines.

A familiar, Pinecone-shaped API. Bring your own vectors, search, filter, and stay compliant.

# pip install vektoria from vektoria import IndexManager mgr = IndexManager("./data") mgr.create_index("contracts", dimension=384) index = mgr.get("contracts") # Upsert your vectors (text lives in metadata) index.upsert([ {"id": "c1", "values": embed("clause de non-concurrence"), "metadata": {"text": "clause de non-concurrence", "source": "contrat.pdf"}}, ]) # Hybrid search: semantic vectors + BM25 keywords hits = index.query(embed("non-concurrence"), top_k=5, hybrid=True, text="non-concurrence") # GDPR: real erasure + export index.delete(filter={"source": "contrat.pdf"}) dump = index.export()
